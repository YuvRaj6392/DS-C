#include<bits/stdc++.h>
using namespace std;
class shop
{
int id;
int price;
public:
void setdata(int val1,int val2)
{
id=val1;
price=val2;
}
void getdata()
{
cout<<"Id= "<<id<<endl;
cout<<"price= "<<price<<endl;
}
};
int main()
{
	int p,q;
	shop *ptr=new shop[3];
	shop *temp=ptr;
	for(int i=0;i<3;i++)
	{
		cin>>p>>q;
		ptr->setdata(p,q);
		ptr++;
	}
	for(int i=0;i<3;i++)
	{
		temp->getdata();
		temp++;
	}
return 0;
}
