#include<stdio.h>
#include<stdlib.h>
struct node
{
int data;
struct node *next;
};
void traversal(struct node *ptr)
{
	while(ptr!=NULL)
	{
		printf("%d\n",ptr->data);
		ptr=ptr->next;
	}
}
void insertingafternode(struct node *ptr,struct node *ptr1)
{
	ptr1->data=99;
	ptr1->next=ptr->next;
	ptr->next=ptr1;
	}
void insertingbeginning(struct node *ptr,struct node *ptr1)
{
	ptr1->data=99;
	ptr1->next=ptr;
}
void insertingatend(struct node *ptr,struct node *ptr1)
{
while(ptr->next!=NULL)
{
	ptr=ptr->next;
}
ptr->next=ptr1;
ptr1->data=43;
ptr1->next=NULL;
}
