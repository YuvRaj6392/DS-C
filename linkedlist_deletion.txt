#include<stdio.h>
#include<stdlib.h>
struct node
{
int data;
struct node *next;
};
void traversal(struct node *ptr)
{
	while(ptr!=NULL)
	{
		printf("%d\n",ptr->data);
		ptr=ptr->next;
	}
}
struct node * insertionbeginning(struct node *head)
{
struct node *ptr;
ptr=(struct node *)malloc(sizeof(struct node));
ptr->next=head;
ptr->data=69;
return ptr;
}
void  insertionafterindex(struct node *head,int index)
{
struct node *p=head;
struct node *ptr=(struct node *)malloc(sizeof(struct node));	
int i=0;
while(i!=index-1)
{
	p=p->next;
	i++;
}
ptr->next=p->next;
ptr->data=63;
p->next=ptr;
}
struct node * deletingfirst(struct node *head)
{
struct node *p1=head;
head=head->next;
free(p1);
return head;
}
void deletingatlast(struct node *head)
{
	struct node *p=head;
	struct node *q=head->next;
	while(q->next!=NULL)
	{
		p=q;
		q=q->next;
	}
	p->next=NULL;
	free(q);
}
void deletingatvalue(struct node *head,int value)
{
	struct node *p=head;
	struct node *q=head->next;
	while(q->data!=value)
	{
		p=q;
		q=q->next;
	}
	p->next=q->next;
	free(q);
}
void deletemiddle(struct node *head,struct node *ptr)
{
	struct node *p=head;
	struct node *q=head->next;
	while(q!=ptr)
	{
		p=q;
		q=q->next;
	}
	p->next=q->next;
	free(q);
}
int main()
{
struct node *head;
struct node *second;
struct node *third;
struct node *fourth;
head=(struct node *)malloc(sizeof(struct node));
second=(struct node *)malloc(sizeof(struct node));
third=(struct node *)malloc(sizeof(struct node));
fourth=(struct node *)malloc(sizeof(struct node));
head->data=23;
head->next=second;
second->data=45; 
second->next=third;
third->data=67;
third->next=fourth;
fourth->data=78;
fourth->next=NULL;
//Insertion at the beginning
head=insertionbeginning(head);
//Insertion after index
insertionafterindex(head,2);
//deleting the first node
head=deletingfirst(head);
//deleting at last
deletingatlast(head);
//deleting a node at a given value
deletingatvalue(head,45);
//deleting at middle
deletemiddle(head,third);
traversal(head);
return 0;
}
