#include<bits/stdc++.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
int find(int ar[],int n)
{
for(int i=0;i<10;i++)
{
	if(ar[i]==n)
	{
		return 1;
	}
}
return 0;
}
int main()
{
int f,n,ar[50]={21,22,45,12,67,87,45,90,11,3};
cout<<"Enter the number to find"<<endl;
cin>>n;
f=find(ar,n);
if(f==1)
{
	cout<<"Found";
}
else
{
	cout<<"Not found";
}
return 0;
}
