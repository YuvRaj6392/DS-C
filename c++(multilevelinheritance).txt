#include<bits/stdc++.h>
using namespace std;
class student
{
protected:
int roll;
public:
void setroll(int n)
{
roll=n;
}
void getroll()
{
		cout<<"Roll number= "<<roll<<endl;
}
};
class marks:public student
{
protected:
float physics;
float maths;
public:
void setmarks(float p,float m)
{
physics=p;
maths=m;
}
void getmarks()
{
		cout<<"physics="<<physics<<endl;
		cout<<"maths="<<maths<<endl;
}
};
class result:public marks
{
public:
void display()
{
getroll();
getmarks();
cout<<"Result="<<(physics+maths)/2<<endl;
}
};
int main()
{
result r1;
r1.setroll(23);
r1.setmarks(23,45);
r1.display();
return 0;
}
