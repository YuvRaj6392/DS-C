#include<stdio.h>
#include<stdlib.h>
struct node
{
int data;
struct node *next;
};
void traversal(struct node *ptr)
{
	while(ptr!=NULL)
	{
		printf("%d\n",ptr->data);
		ptr=ptr->next;
	}
}
int main()
{
struct node *head;
struct node *second;
struct node *third;
struct node *fourth;
head=(struct node *)malloc(sizeof(struct node));
second=(struct node *)malloc(sizeof(struct node));
third=(struct node *)malloc(sizeof(struct node));
fourth=(struct node *)malloc(sizeof(struct node));
head->data=23;
head->next=second;
second->data=45;
second->next=third;
third->data=67;
third->next=fourth;
fourth->data=78;

}
