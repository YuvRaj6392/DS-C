#include<stdio.h>
#include<stdlib.h>
struct node
{
int data;
struct node *next;
};
void traversal(struct node *ptr)
{
	while(ptr!=NULL)
	{
		printf("%d\n",ptr->data);
		ptr=ptr->next;
	}
}
struct node * insertionbeginning(struct node *head)
{
struct node *ptr;
ptr=(struct node *)malloc(sizeof(struct node));
ptr->next=head;
ptr->data=69;
return ptr;
}
void  insertionafterindex(struct node *head,int index)
{
struct node *p=head;
struct node *ptr=(struct node *)malloc(sizeof(struct node));	
int i=0;
while(i!=index-1)
{
	p=p->next;
	i++;
}
ptr->next=p->next;
ptr->data=63;
p->next=ptr;
}
int main()
{
struct node *head;
struct node *second;
struct node *third;
struct node *fourth;
head=(struct node *)malloc(sizeof(struct node));
second=(struct node *)malloc(sizeof(struct node));
third=(struct node *)malloc(sizeof(struct node));
fourth=(struct node *)malloc(sizeof(struct node));
head->data=23;
head->next=second;
second->data=45; 
second->next=third;
third->data=67;
third->next=fourth;
fourth->data=78;
fourth->next=NULL;
//Insertion at the beginning
head=insertionbeginning(head);
//Insertion after index
insertionafterindex(head,2);
traversal(head);
return 0;
}
